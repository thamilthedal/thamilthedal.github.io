2. Grid Discretization Uniform Meshing (Calculating Fc,Cc,Fw and Cw)

In [3]:
Cw = []
Fw = []
Fc = []
Cc = []
(imax,jmax) = Node
for i in range(ND):

    Cw.append([0]*(Node[i]-1))
    Fc.append([0]*(Node[i]-1))
    Cc.append([0]*Node[i])
    Fw.append([0]*(Elem[i]))
    
    #Facewidths for Uniform Meshing
    for a in range(Elem[i]):
        Fw[i][a] = L[i]/Elem[i]
    
    #Cell Widths, Cell Centers and Face Centers
    Cc[i][0] = 0
    Fc[i][0] = 0
    Cc[i][Elem[i]+1] = L[i]
    for b in range(1,Elem[i]+1):
        Fc[i][b] = Fc[i][b-1]+Fw[i][b-1]
        Cc[i][b] = ((Fc[i][b]+Fc[i][b-1])/2)
        Cw[i][b-1] = Cc[i][b] - Cc[i][b-1]
    Cw[i][Elem[i]] = L[i] - Cc[i][Elem[i]]

Solution Methodology¶

In [25]:
# Flux Based Solution
t=0
Time=[]
Residual=[]
while (epsi_st>10e-4):
    boundary()
    T_old= copy.deepcopy(T)
    
    for i in range(1,imax-1):
        for j in range(1,jmax-1):
            
            # Calculation of Heat-flux
            qx[i+1,j]=-k*(T[i+1,j]-T[i,j])/Cw[0][i]
            qx[i-1,j]=-k*(T[i,j]-T[i-1,j])/Cw[0][i-1]
            qy[i,j+1]=-k*(T[i,j+1]-T[i,j])/Cw[1][j]
            qy[i,j-1]=-k*(T[i,j]-T[i,j-1])/Cw[1][j-1]
            
            # delX and delY
            delX = Fw[0][i-1]
            delY = Fw[1][j-1]
            
            # Conduction Flux
            Qcond[i,j]=((qx[i-1,j]-qx[i+1,j])*delY)+((qy[i,j-1]-qy[i,j+1])*delX)
            Qgen=qgen*delX*delY
            
            # Calculation of Temperature
            T[i,j]=T_old[i,j]+(delT/(rho*cp*delX*delY))*(Qcond[i,j]+Qgen)
            
            # Error
            err[i-1,j-1]=abs((((T[i,j]-T_old[i,j])/(T[0,0]-T[imax-1,jmax-1]))/Fo_st))
    
    epsi_st=max(max(err.reshape(1,err.size)))
    if(t%10 == 0):
        print(t,epsi_st)
    t=t+1
print("The Solution is converged\n")
